# 介绍

基于element-ui实现，通过配置生成的表单，支持多种场景

# 快速使用

# npm方式

# 请先确保正确安装并使用了 element-ui
npm install --save tyro-form-builder
1
2
<template>
  <form-builder :form-data="formData" />
</template>

<script>
  import FormBuilder from 'tyro-form-builder'

  export default {
    components: {
      FormBuilder,
    },
    data() {
      return {
        formData: {
          list: [{
            type: 'input',
            model: 'name',
            label: '名字'
          }]
        }
      }
    }
  }
</script>
