    # 上传文件 upload

      <template>
  <div class="demo-box">
    <form-builder :form-data="formData" />
  </div>
</template>

<script>
export default {
  data() {
    return {
      formData: {
        list: [{
          type: 'upload',
          model: 'upload',
          label: '上传',
          attrs: {
            class: 'avatar-uploader',
            action: ''
          },
          content: <i class="el-icon-plus avatar-uploader-icon"></i>
        }]
      }
    }
  }
}
</script>
